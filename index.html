Q3. Perfect Number Check.

Ans.
                var Perfect_Check = (N) => {
                    let div=[];
                    for(let i=0;i< 100000;i++){
                      if(N%i===0){
                        div.push(i)
                      }
                    }
                    let sum=0;
                    for(let i=0;i< div.length;i++){
                      sum+=div[i];
                    }
                    if(sum==N){
                      return 'YES'
                    }else{
                      return 'NO'
                    }
                  };

Q4. Reverse a Number.

Ans.
                var Reverse_Number = (N) =>{
                    let m=N.toString();
                    let done='';
                    for(let i=m.length-1;i>=0;i--){
                      done+=m[i]
                    }
                    return parseInt(done);
                  };
